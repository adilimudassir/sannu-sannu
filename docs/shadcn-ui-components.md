# shadcn/ui Components Guide
## Sannu-Sannu SaaS Platform

### Overview

This document provides comprehensive guidance on using shadcn/ui components within the Sannu-Sannu platform. shadcn/ui provides a collection of copy-and-paste components built using Radix UI and Tailwind CSS, ensuring accessibility, consistency, and modern design patterns.

### Installation & Setup

#### Prerequisites
- React 19.x with TypeScript 5.x
- Tailwind CSS 4.x
- Vite 6.x

#### Installation
```bash
# Initialize shadcn/ui (latest version)
pnpm dlx shadcn@latest init

# Install core components
pnpm dlx shadcn@latest add button card input dialog progress badge form table toast
pnpm dlx shadcn@latest add tabs select textarea label dropdown-menu
pnpm dlx shadcn@latest add alert-dialog sheet popover command
```

#### Configuration
```json
// components.json
{
  "$schema": "https://ui.shadcn.com/schema.json",
  "style": "new-york",
  "rsc": false,
  "tsx": true,
  "tailwind": {
    "config": "tailwind.config.js",
    "css": "resources/css/app.css",
    "baseColor": "neutral",
    "cssVariables": true,
    "prefix": ""
  },
  "aliases": {
    "components": "@/Components",
    "utils": "@/lib/utils",
    "ui": "@/Components/ui"
  }
}
```

### Why shadcn/ui?

- **Accessibility First**: Built on Radix UI primitives with full keyboard navigation and screen reader support
- **Customizable**: Components are copied into your project, allowing full customization
- **Type Safe**: Full TypeScript support with proper type definitions
- **Modern Design**: Clean, modern aesthetic that works well for SaaS applications
- **Theme Integration**: Seamlessly integrates with our configurable theming system
- **Developer Experience**: Excellent documentation and easy integration

### Theme Integration

All shadcn/ui components automatically inherit colors from the configured theme through CSS custom properties. See the [Simple Theming](./simple-theming.md) documentation for color configuration and customization options.

### Form Components

```tsx
import { Input } from '@/Components/ui/input';
import { Label } from '@/Components/ui/label';
import { Textarea } from '@/Components/ui/textarea';
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from '@/Components/ui/select';

// Form Example with Inertia.js
export default function ProjectForm() {
  const { data, setData, post, processing, errors } = useForm({
    name: '',
    description: '',
    monthly_amount: '',
    duration_months: '',
  });

  return (
    <form onSubmit={(e) => { e.preventDefault(); post('/projects'); }} className="space-y-4">

        {errors.description && <p className="text-sm text-destructive">{errors.description}</p>}

      <div className="space-y-2">

        <Select onValueChange={(value) => setData('duration_months', value)}>
```

### Dialog/Modal Components

```tsx
import { Dialog, DialogContent, DialogDescription, DialogFooter, DialogHeader, DialogTitle, DialogTrigger } from '@/Components/ui/dialog';

// Join Project Modal
export function JoinProjectDialog({ project }: { project: Project }) {
  const [open, setOpen] = useState(false);
  const { data, setData, post, processing } = useForm({
    duration_months: '3',
    payment_type: 'monthly',
  });

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    post(`/projects/${project.id}/join`, {
      onSuccess: () => setOpen(false),
    });
  };

  return (
    <Dialog open={open} onOpenChange={setOpen}>
      <DialogTrigger asChild>
        <Button>Join Project</Button>
      </DialogTrigger>
      <DialogContent className="sm:max-w-[425px]">
        <DialogHeader>
          <DialogTitle>Join {project.name}</DialogTitle>
          <DialogDescription>
            Choose your contribution plan for this project.
          </DialogDescription>
        </DialogHeader>
        <form onSubmit={handleSubmit} className="space-y-4">
          <div className="space-y-2">
            <Label>Duration</Label>
            <Select value={data.duration_months} onValueChange={(value) => setData('duration_months', value)}>
              <SelectTrigger>
                <SelectValue />
              </SelectTrigger>
              <SelectContent>
                <SelectItem value="3">3 Months</SelectItem>
                <SelectItem value="6">6 Months</SelectItem>
                <SelectItem value="12">12 Months</SelectItem>
              </SelectContent>
            </Select>
          </div>
          <DialogFooter>
            <Button type="button" variant="outline" onClick={() => setOpen(false)}>
              Cancel
            </Button>
            <Button type="submit" disabled={processing}>
              {processing ? 'Joining...' : 'Join Project'}
            </Button>
          </DialogFooter>
        </form>
      </DialogContent>
    </Dialog>
  );
}
```

### Progress Ring Component

```tsx
// resources/js/Components/ui/progress-ring.tsx
import React from 'react';

interface ProgressRingProps {
  progress: number;
  size?: number;
  strokeWidth?: number;
  className?: string;
}

export function ProgressRing({ 
  progress, 
  size = 120, 
  strokeWidth = 8, 
  className = '' 
}: ProgressRingProps) {
  const radius = (size - strokeWidth) / 2;
  const circumference = radius * 2 * Math.PI;
  const strokeDasharray = `${circumference} ${circumference}`;
  const strokeDashoffset = circumference - (progress / 100) * circumference;

  return (
    <div className={`relative ${className}`}>
      <svg
        className="transform -rotate-90"
        width={size}
        height={size}
      >
        <circle
          className="text-muted"
          strokeWidth={strokeWidth}
          stroke="currentColor"
          fill="transparent"
          r={radius}
          cx={size / 2}
          cy={size / 2}
        />
        <circle
          className="text-primary transition-all duration-300 ease-in-out"
          strokeWidth={strokeWidth}
          strokeDasharray={strokeDasharray}
          strokeDashoffset={strokeDashoffset}
          strokeLinecap="round"
          stroke="currentColor"
          fill="transparent"
          r={radius}
          cx={size / 2}
          cy={size / 2}
        />
      </svg>
      <div className="absolute inset-0 flex items-center justify-center">
        <span className="text-xl font-semibold">{progress.toFixed(1)}%</span>
      </div>
    </div>
  );
}
```

### Payment Status Badge

```tsx
// resources/js/Components/Payment/PaymentStatusBadge.tsx
import { Badge } from '@/Components/ui/badge';
import { CheckCircle, XCircle, Clock, AlertCircle } from 'lucide-react';

interface PaymentStatusBadgeProps {
  status: 'success' | 'failed' | 'pending' | 'cancelled';
}

export function PaymentStatusBadge({ status }: PaymentStatusBadgeProps) {
  const config = {
    success: {
      variant: 'default' as const,
      icon: CheckCircle,
      label: 'Success',
      className: 'bg-green-100 text-green-800 hover:bg-green-100',
    },
    failed: {
      variant: 'destructive' as const,
      icon: XCircle,
      label: 'Failed',
      className: '',
    },
    pending: {
      variant: 'secondary' as const,
      icon: Clock,
      label: 'Pending',
      className: 'bg-yellow-100 text-yellow-800 hover:bg-yellow-100',
    },
    cancelled: {
      variant: 'outline' as const,
      icon: AlertCircle,
      label: 'Cancelled',
      className: '',
    },
  };

  const { variant, icon: Icon, label, className } = config[status];

  return (
    <Badge variant={variant} className={`flex items-center gap-1 ${className}`}>
      <Icon className="h-3 w-3" />
      {label}
    </Badge>
  );
}
```

### Dark Mode Support

```tsx
// resources/js/Components/ThemeProvider.tsx
import React, { createContext, useContext, useEffect, useState } from 'react';

type Theme = 'dark' | 'light' | 'system';

type ThemeProviderProps = {
  children: React.ReactNode;
  defaultTheme?: Theme;
  storageKey?: string;
};

const ThemeProviderContext = createContext<{
  theme: Theme;
  setTheme: (theme: Theme) => void;
}>({
  theme: 'system',
  setTheme: () => null,
});

export function ThemeProvider({
  children,
  defaultTheme = 'system',
  storageKey = 'sannu-sannu-theme',
  ...props
}: ThemeProviderProps) {
  const [theme, setTheme] = useState<Theme>(
    () => (localStorage.getItem(storageKey) as Theme) || defaultTheme
  );

  useEffect(() => {
    const root = window.document.documentElement;
    root.classList.remove('light', 'dark');

    if (theme === 'system') {
      const systemTheme = window.matchMedia('(prefers-color-scheme: dark)').matches
        ? 'dark'
        : 'light';
      root.classList.add(systemTheme);
      return;
    }

    root.classList.add(theme);
  }, [theme]);

  const value = {
    theme,
    setTheme: (theme: Theme) => {
      localStorage.setItem(storageKey, theme);
      setTheme(theme);
    },
  };

  return (
    <ThemeProviderContext.Provider {...props} value={value}>
      {children}
    </ThemeProviderContext.Provider>
  );
}

export const useTheme = () => {
  const context = useContext(ThemeProviderContext);
  if (context === undefined)
    throw new Error('useTheme must be used within a ThemeProvider');
  return context;
};
```

### Accessibility Guidelines

1. **Always use proper semantic HTML**
2. **Include proper ARIA labels and descriptions**
3. **Ensure keyboard navigation works**
4. **Maintain proper color contrast ratios**
5. **Use shadcn/ui components which are built with accessibility in mind**

### Performance Considerations

1. **Lazy load heavy components**
2. **Use React.memo for expensive renders**
3. **Optimize bundle size by importing only needed components**
4. **Use proper TypeScript types for better tree shaking**

This comprehensive guide ensures consistent, accessible, and maintainable UI components throughout the Sannu-Sannu platform using shadcn/ui.
